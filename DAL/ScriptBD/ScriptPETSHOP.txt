CREATE PROCEDURE [dbo].[sp_ValidarUsuario]
    @nombre NVARCHAR(50),
    @pass NVARCHAR(128)
AS
BEGIN
    SELECT Id, Nombre, Permiso 
    FROM USUARIO 
    WHERE Nombre = @nombre AND LOWER(Password) = LOWER(@pass)
END
